190 void Agent :: Start(){
191 
192  // Create agent logs file if required
193  if(save_agent_logs) {
194  // Name agent log file accordingly to the agent_id
195  sprintf(own_file_path,"%s_A%d_%s.txt","../output/logs_output", agent_id, wlan_code.c_str());
196  remove(own_file_path);
197  output_log_file = fopen(own_file_path, "at");
198  agent_logger.save_logs = save_agent_logs;
199  agent_logger.file = output_log_file;
200  agent_logger.SetVoidHeadString();
201  }
202 
203  if(save_agent_logs) fprintf(agent_logger.file,"%.18f;A%d;%s;%s Start()\n",
204  SimTime(), agent_id, LOG_B00, LOG_LVL1);
205 
206  if(centralized_flag) {
207  // --- Do nothing ---
208  // In case of being centralized, wait for a request from the controller
209  } else {
210  // Generate the first request, to be triggered after "time_between_requests"
211  // *** We generate here the first request in order to obtain the AP's configuration
212  double extra_wait_test = 0.005 * (double) agent_id;
213  trigger_request_information_to_ap.Set(fix_time_offset(SimTime() + time_between_requests + extra_wait_test,13,12));
214  }
215
